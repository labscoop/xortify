278 if (!function_exists("xortify_xml2array")) {
279  function xortify_xml2array($contents, $get_attributes=1, $priority = 'tag') {
280  if(!$contents) return array();
281 
282  if(!function_exists('xml_parser_create')) {
283  return array();
284  }
285 
286  //Get the XML parser of PHP - PHP must have this module for the parser to work
287  $parser = xml_parser_create('');
288  xml_parser_set_option($parser, XML_OPTION_TARGET_ENCODING, "UTF-8"); # http://minutillo.com/steve/weblog/2004/6/17/php-xml-and-character-encodings-a-tale-of-sadness-rage-and-data-loss
289  xml_parser_set_option($parser, XML_OPTION_CASE_FOLDING, 0);
290  xml_parser_set_option($parser, XML_OPTION_SKIP_WHITE, 1);
291  xml_parse_into_struct($parser, trim($contents), $xml_values);
292  xml_parser_free($parser);
293 
294  if(!$xml_values) return;//Hmm...
295 
296  //Initializations
297  $xml_array = array();
298  $parents = array();
299  $opened_tags = array();
300  $arr = array();
301 
302  $current = &$xml_array; //Refference
303 
304  //Go through the tags.
305  $repeated_tag_index = array();//Multiple tags with same name will be turned into an array
306  foreach($xml_values as $data) {
307  unset($attributes,$value);//Remove existing values, or there will be trouble
308 
309  //This command will extract these variables into the foreach scope
310  // tag(string), type(string), level(int), attributes(array).
311  extract($data);//We could use the array by itself, but this cooler.
312 
313  $result = array();
314  $attributes_data = array();
315 
316  if(isset($value)) {
317  if($priority == 'tag') $result = $value;
318  else $result['value'] = $value; //Put the value in a assoc array if we are in the 'Attribute' mode
319  }
320 
321  //Set the attributes too.
322  if(isset($attributes) and $get_attributes) {
323  foreach($attributes as $attr => $val) {
324  if($priority == 'tag') $attributes_data[$attr] = $val;
325  else $result['attr'][$attr] = $val; //Set all the attributes in a array called 'attr'
326  }
327  }
328 
329  //See tag status and do the needed.
330  if($type == "open") {//The starting of the tag '<tag>'
331  $parent[$level-1] = &$current;
332  if(!is_array($current) or (!in_array($tag, array_keys($current)))) { //Insert New tag
333  $current[$tag] = $result;
334  if($attributes_data) $current[$tag. '_attr'] = $attributes_data;
335  $repeated_tag_index[$tag.'_'.$level] = 1;
336 
337  $current = &$current[$tag];
338 
339  } else { //There was another element with the same tag name
340 
341  if(isset($current[$tag][0])) {//If there is a 0th element it is already an array
342  $current[$tag][$repeated_tag_index[$tag.'_'.$level]] = $result;
343  $repeated_tag_index[$tag.'_'.$level]++;
344  } else {//This section will make the value an array if multiple tags with the same name appear together
345  $current[$tag] = array($current[$tag],$result);//This will combine the existing item and the new item together to make an array
346  $repeated_tag_index[$tag.'_'.$level] = 2;
347 
348  if(isset($current[$tag.'_attr'])) { //The attribute of the last(0th) tag must be moved as well
349  $current[$tag]['0_attr'] = $current[$tag.'_attr'];
350  unset($current[$tag.'_attr']);
351  }
352 
353  }
354  $last_item_index = $repeated_tag_index[$tag.'_'.$level]-1;
355  $current = &$current[$tag][$last_item_index];
356  }
357 
358  } elseif($type == "complete") { //Tags that ends in 1 line '<tag />'
359  //See if the key is already taken.
360  if(!isset($current[$tag])) { //New Key
361  $current[$tag] = $result;
362  $repeated_tag_index[$tag.'_'.$level] = 1;
363  if($priority == 'tag' and $attributes_data) $current[$tag. '_attr'] = $attributes_data;
364 
365  } else { //If taken, put all things inside a list(array)
366  if(isset($current[$tag][0]) and is_array($current[$tag])) {//If it is already an array...
367 
368  // ...push the new element into that array.
369  $current[$tag][$repeated_tag_index[$tag.'_'.$level]] = $result;
370 
371  if($priority == 'tag' and $get_attributes and $attributes_data) {
372  $current[$tag][$repeated_tag_index[$tag.'_'.$level] . '_attr'] = $attributes_data;
373  }
374  $repeated_tag_index[$tag.'_'.$level]++;
375 
376  } else { //If it is not an array...
377  $current[$tag] = array($current[$tag],$result); //...Make it an array using using the existing value and the new value
378  $repeated_tag_index[$tag.'_'.$level] = 1;
379  if($priority == 'tag' and $get_attributes) {
380  if(isset($current[$tag.'_attr'])) { //The attribute of the last(0th) tag must be moved as well
381 
382  $current[$tag]['0_attr'] = $current[$tag.'_attr'];
383  unset($current[$tag.'_attr']);
384  }
385 
386  if($attributes_data) {
387  $current[$tag][$repeated_tag_index[$tag.'_'.$level] . '_attr'] = $attributes_data;
388  }
389  }
390  $repeated_tag_index[$tag.'_'.$level]++; //0 and 1 index is already taken
391  }
392  }
393 
394  } elseif($type == 'close') { //End of tag '</tag>'
395  $current = &$parent[$level-1];
396  }
397  }
398 
399  return($xml_array);
400  }
