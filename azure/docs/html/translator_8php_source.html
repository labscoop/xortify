92  $ch = curl_init();

95  //Set the HTTP HEADER Fields.

97  //CURLOPT_RETURNTRANSFER- TRUE to return the transfer as a string of the return value of curl_exec().
98  curl_setopt ($ch, CURLOPT_RETURNTRANSFER, TRUE);
99  //CURLOPT_SSL_VERIFYPEER- Set FALSE to stop cURL from verifying the peer's certificate.
100  curl_setopt ($ch, CURLOPT_SSL_VERIFYPEER, False);
101  //Execute the cURL session.
102  $json = curl_exec($ch);
103  //Get the Error Code returned by Curl.
104  $json_errono = curl_errno($ch);
105  // create a connection to the API endpoint
106  curl_close($ch);
107  // decode the response data
108  $data = json_decode($json, true);
109 
110  // ensure the returned data is valid
111  if (!is_array($data) || !array_key_exists('data', $data)) {
112  $GLOBAL['translation'] = false;

174  //Set HTTP POST Request.
175  curl_setopt($ch, CURLOPT_POST, TRUE);

179  curl_setopt ($ch, CURLOPT_RETURNTRANSFER, TRUE);
